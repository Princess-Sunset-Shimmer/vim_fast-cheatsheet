## open save and quit:
        vim file.s             /* open file.s */
        :w                     /* save file */
        :q                     /* quit */
        :q!                    /* quit without save */
        :wq                    /* save and quit */
## mode switch:
        esc                    /* view mode */
        i                      /* inset mode */
        /                      /* search mode */
        :                      /* command mode */
        v                      /* selection mode */
## cursor control:
        h, j, k, l             /* ⬅️, ⬆️, ⬇️, ➡️ that every vim user knows */
        w                      /* forword */
        b                      /* backword */
        $                      /* end */
        0                      /* home */
        shift + g              /* tail of file */
        g + g                  /* mane of file */
